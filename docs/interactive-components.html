Interactive Components

Shiny UI elements styled with vvnthemes.

Styled Shiny UI components for VVN dashboards — filters, sliders, buttons, and KPI cards. These are used in the Dashboard Template only, not in VVN Insights documents.

Required packages

library(vvnthemes)
library(shiny)
library(bslib)

For layout rules and filter panel standards, see Dashboard Standards →

Which component should I use?

I need… Component Function
Let users pick a category or region Dropdown filter vvn_filter()
Let users select a year range Range slider vvn_slider()
Trigger filtering or reset state Action button vvn_button()
Show a headline metric with trend KPI metric card vvn_kpi_card()

Quick start — complete sidebar

A standard VVN sidebar with two filters, a slider, and Apply / Reset buttons:

sidebar <- bslib::sidebar(
  title = "Filters", width = 260, bg = "#F7F7F7",

  vvn_filter("region_sel", "Region",
             choices  = c("All", "Rural", "Urban"),
             selected = "All", multiple = TRUE),

  vvn_slider("year_range", "Year Range",
             min = 2015, max = 2023, value = c(2015, 2023)),

  hr(),

  vvn_button("apply_btn", "Apply Filters",
             icon = bslib::bs_icon("funnel-fill"), style = "primary"),
  tags$br(),
  vvn_button("reset_btn", "Reset",
             icon = bslib::bs_icon("arrow-counterclockwise"), style = "outline")
)

vvn_filter() — dropdown filter

Use when Letting users select one or more categories (region, variable, topic). Produces a styled dropdown with a bold label. 
vvn_filter(
  id       = "region_sel",
  label    = "Select Region",
  choices  = c("All" = "All", "Rural" = "Rural", "Urban" = "Urban"),
  selected = "All",
  multiple = TRUE
)

Parameters

Parameter Type Default Description
id character required Shiny input ID
label character required Display label shown above the dropdown
choices named vector required Options — name = displayed text, value = code
selected character NULL Initially selected value(s)
multiple logical FALSE Allow multiple selections

Access in server

input$region_sel   # character vector of selected values

vvn_slider() — range slider

Use when Letting users set a numeric range or select a single year. Produces a styled single-handle or two-handle slider. 
# Range slider (two handles)
vvn_slider("year_range", "Year Range", min = 2015, max = 2023, value = c(2015, 2023))

# Single-value slider
vvn_slider("year_sel", "Year", min = 2015, max = 2023, value = 2023)

Parameters

Parameter Type Description
id character Shiny input ID
label character Display label
min numeric Minimum value
max numeric Maximum value
value numeric or length-2 vector Initial value(s) — use c(min, max) for a range slider

Access in server

# Range: input$year_range is a length-2 vector [low, high]
filtered_data <- reactive({
  df |> filter(year >= input$year_range[1], year <= input$year_range[2])
})

vvn_button() — action button

Use when Triggering a filter action or resetting state. Use “primary” (maroon) for Apply Filters and “outline” for Reset. 
# Primary (maroon) — Apply Filters
vvn_button("apply_btn", "Apply Filters",
           icon = bslib::bs_icon("funnel-fill"), style = "primary")

# Outline — Reset
vvn_button("reset_btn", "Reset",
           icon = bslib::bs_icon("arrow-counterclockwise"), style = "outline")

Parameters

Parameter Type Default Description
id character required Shiny input ID
label character required Button text
icon icon NULL Optional icon from bslib::bs_icon()
style character "primary" "primary" (maroon) · "secondary" · "outline"

vvn_kpi_card() — KPI metric card

Use when Displaying a headline metric at the top of a dashboard. Use three per row with maroon, orange, and navy color schemes. 
vvn_kpi_card(
  value        = "42%",
  label        = "Employment Rate",
  trend        = "up",
  trend_text   = "+3.1% vs 2022",
  color_scheme = "maroon"
)

Parameters

Parameter Type Default Description
value character or numeric required The large metric value displayed
label character required Short description of the metric
trend character NULL "up" · "down" · "flat" · NULL (no indicator)
trend_text character NULL Supporting text (e.g., “+3% vs 2022”)
color_scheme character "maroon" "maroon" · "orange" · "navy"

Three-card layout in a Shiny app

# Server — compute values from filtered data
output$kpi_mean   <- renderUI({
  vvn_kpi_card(round(mean(filtered()$value, na.rm = TRUE), 1),
               "Average Value", color_scheme = "maroon")
})
output$kpi_n      <- renderUI({
  vvn_kpi_card(nrow(filtered()), "Counties", color_scheme = "orange")
})
output$kpi_change <- renderUI({
  d  <- filtered()
  ch <- round(mean(d$value[d$year == max(d$year)], na.rm = TRUE) -
              mean(d$value[d$year == min(d$year)], na.rm = TRUE), 1)
  vvn_kpi_card(
    value        = paste0(if (ch >= 0) "+" else "", ch),
    label        = paste0("Change (", min(d$year), "–", max(d$year), ")"),
    trend        = if (ch >= 0) "up" else "down",
    color_scheme = "navy"
  )
})

# UI — three equal columns
bslib::layout_columns(
  col_widths = c(4, 4, 4),
  uiOutput("kpi_mean"),
  uiOutput("kpi_n"),
  uiOutput("kpi_change")
)

Standard reactive filter pattern

The recommended server pattern for all VVN dashboards — filtering triggered by the Apply button, Reset restores defaults:

server <- function(input, output, session) {

  filtered <- reactive({
    req(input$year_range)
    d <- app_data

    if (!("All" %in% (input$region_sel %||% "All")))
      d <- d |> filter(region %in% input$region_sel)

    d |>
      filter(year >= input$year_range[1], year <= input$year_range[2]) |>
      mutate(display_value = .data[[input$variable_sel %||% "value"]])

  }) |> bindEvent(input$apply_btn, ignoreNULL = FALSE)

  observeEvent(input$reset_btn, {
    updateSelectInput(session, "region_sel",   selected = "All")
    updateSliderInput(session, "year_range",   value    = c(2015, 2023))
    updateSelectInput(session, "variable_sel", selected = "value")
  })
}

Minimal Shiny app with sidebar

A complete minimal app — one filter, one slider, Apply + Reset, and a chart output:

library(shiny)
library(bslib)
library(ggplot2)
library(vvnthemes)

set_vvn_defaults()

ui <- page_sidebar(
  title = "VVN Dashboard",
  sidebar = sidebar(
    title = "Filters", width = 260, bg = "#F7F7F7",
    vvn_filter("region_sel", "Region",
               choices  = c("All", "Rural", "Urban"),
               selected = "All"),
    vvn_slider("year_sel", "Year", min = 2018, max = 2023, value = 2023),
    hr(),
    vvn_button("apply_btn", "Apply",  style = "primary"),
    tags$br(),
    vvn_button("reset_btn", "Reset",  style = "outline")
  ),
  plotOutput("chart")
)

server <- function(input, output, session) {
  filtered <- reactive({
    d <- app_data
    if (input$region_sel != "All") d <- d |> filter(region == input$region_sel)
    d |> filter(year == input$year_sel)
  }) |> bindEvent(input$apply_btn, ignoreNULL = FALSE)

  observeEvent(input$reset_btn, {
    updateSelectInput(session, "region_sel", selected = "All")
    updateSliderInput(session, "year_sel",   value    = 2023)
  })

  output$chart <- renderPlot({
    ggplot(filtered(), aes(x = value, y = reorder(region, value))) +
      geom_col(width = 0.65) +
      vvn_source("VVN Analysis") +
      remove_ticks()
  })
}

shinyApp(ui, server)
