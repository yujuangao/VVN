Dashboard

Scaffold and build a complete VVN Shiny dashboard.

The create_vvn_dashboard() function scaffolds a complete, ready-to-run R Shiny application in one command. The dashboard runs immediately with placeholder data — you only replace the data source and labels.

Create the project

library(vvnthemes)
create_vvn_dashboard("my-dashboard", title = "Virginia Housing Dashboard")
shiny::runApp("my-dashboard")

Project structure

my-dashboard/
├── app.R           # Main Shiny application — UI and server
├── R/
│   └── data_prep.R # Data loading and transformation
└── www/
    └── vvn.css     # VVN brand styles (do not edit)

What each file does

app.R is the complete Shiny application. It defines the UI (page layout, navbar, sidebar, tabs, cards) and the server (reactive data, outputs for charts, table, and map). The file is pre-wired — all inputs connect to all outputs automatically.

R/data_prep.R is the only file you need to edit to use your own data. It loads and cleans the data and defines app_data, which the rest of the app reads. Your data frame requires at minimum: region (character), year (integer), and value (numeric).

www/vvn.css is the compiled VVN brand stylesheet. Do not edit this file.

Tab structure

Tab Contents
Overview Sidebar filters · KPI cards · trend chart · bar chart · histogram
Map Virginia county-level Leaflet choropleth
Data Searchable DT table with CSV and Excel download
About Project description · data sources · methods · contact

Section 7 — Server: reactive data and reset

Filtered reactive dataset that responds to Apply Filters; reset observer to restore defaults.

# ══════════════════════════════════════════════════════════════════════════════
# Server
# ══════════════════════════════════════════════════════════════════════════════
server <- function(input, output, session) {

  # ── Null-coalescing helper ─────────────────────────────────────────────────
  `%||%` <- function(x, y) if (!is.null(x)) x else y

  # ── Filtered data ──────────────────────────────────────────────────────────
  filtered <- reactive({
    req(input$year_range)
    d <- app_data

    if (!("All" %in% (input$region_sel %||% "All")) &&
        length(input$region_sel) > 0) {
      d <- d |> filter(region %in% input$region_sel)
    }

    d |>
      filter(year >= input$year_range[1], year <= input$year_range[2]) |>
      mutate(display_value = .data[[input$variable_sel %||% "value"]])

  }) |> bindEvent(input$apply_btn, ignoreNULL = FALSE)

  # ── Reset ──────────────────────────────────────────────────────────────────
  observeEvent(input$reset_btn, {
    updateSelectInput(session, "region_sel",   selected = "All")
    updateSliderInput(session, "year_range",   value    = c(2015, 2023))
    updateSelectInput(session, "variable_sel", selected = "value")
  })

Section 8 — Server: KPI cards

Three renderUI outputs producing maroon, orange, and navy VVN KPI cards.

  # ── KPI cards ──────────────────────────────────────────────────────────────
  output$kpi_n <- renderUI({
    n <- dplyr::n_distinct(filtered()$region)
    vvn_kpi_card(paste0(n, " Region", if (n != 1) "s"), "In Selection",
                  color_scheme = "maroon")
  })

  output$kpi_mean <- renderUI({
    m <- round(mean(filtered()$display_value, na.rm = TRUE), 1)
    vvn_kpi_card(m, "Average Value", color_scheme = "orange")
  })

  output$kpi_change <- renderUI({
    d  <- filtered()
    v1 <- mean(d$display_value[d$year == min(d$year)], na.rm = TRUE)
    v2 <- mean(d$display_value[d$year == max(d$year)], na.rm = TRUE)
    ch <- round(v2 - v1, 1)
    vvn_kpi_card(
      value        = paste0(if (ch >= 0) "+" else "", ch),
      label        = paste0("Change (", min(d$year), "\u2013", max(d$year), ")"),
      trend        = if (ch >= 0) "up" else "down",
      trend_text   = paste0(abs(round(ch / v1 * 100)), "% vs start"),
      color_scheme = "navy"
    )
  })

Section 9 — Server: charts

Trend line chart, horizontal bar chart, and histogram — all using theme_vvn() and VVN color scales.

  # ── Trend chart ────────────────────────────────────────────────────────────
  output$trend_chart <- renderPlot({
    req(nrow(filtered()) > 0)
    filtered() |>
      group_by(year, region) |>
      summarise(v = mean(display_value, na.rm = TRUE), .groups = "drop") |>
      ggplot(aes(x = year, y = v, color = region, group = region)) +
      geom_line(linewidth = 1.3) + geom_point(size = 2.8, alpha = 0.9) +
      scale_color_vvn("main") + theme_vvn() +
      labs(title    = "Trend by Region",
           subtitle = paste0(input$year_range[1], "\u2013", input$year_range[2]),
           x = "Year", y = "Value", color = "Region",
           caption  = "Source: [Replace with source]")
  }, res = 110)

  # ── Bar chart ──────────────────────────────────────────────────────────────
  output$bar_chart <- renderPlot({
    req(nrow(filtered()) > 0)
    filtered() |>
      filter(year == max(year)) |>
      group_by(region) |>
      summarise(v = mean(display_value, na.rm = TRUE), .groups = "drop") |>
      arrange(v) |> mutate(region = factor(region, levels = region)) |>
      ggplot(aes(x = v, y = region, fill = region)) +
      geom_col(width = 0.65, show.legend = FALSE) +
      geom_text(aes(label = round(v, 1)), hjust = -0.2, size = 3.2) +
      scale_fill_vvn("main") +
      scale_x_continuous(expand = expansion(mult = c(0, .18))) +
      theme_vvn(grid = "x") + remove_ticks() +
      labs(title = paste("Comparison \u00b7", max(filtered()$year)),
           x = "Value", y = NULL)
  }, res = 110)

  # ── Histogram ─────────────────────────────────────────────────────────────
  output$hist_chart <- renderPlot({
    req(nrow(filtered()) > 0)
    m <- mean(filtered()$display_value, na.rm = TRUE)
    ggplot(filtered(), aes(x = display_value)) +
      geom_histogram(bins = 18, fill = "#861F41", color = "#FFFFFF", alpha = 0.9) +
      geom_vline(xintercept = m, color = "#E5751F", linewidth = 1.2, linetype = "dashed") +
      annotate("text", x = m + 0.3, y = Inf, vjust = 2, hjust = 0, size = 3,
               label = paste0("Mean: ", round(m, 1)), color = "#E5751F") +
      theme_vvn() +
      labs(title = "Distribution", x = "Value", y = "Count")
  }, res = 110)

**`www/vvn.css`** is the compiled VVN brand stylesheet. Do not edit this file.

### Section 7 — Server: reactive data and reset

Filtered reactive dataset that responds to Apply Filters; reset observer to restore defaults.

```r
# ══════════════════════════════════════════════════════════════════════════════
# Server
# ══════════════════════════════════════════════════════════════════════════════
server <- function(input, output, session) {

  # ── Null-coalescing helper ─────────────────────────────────────────────────
  `%||%` <- function(x, y) if (!is.null(x)) x else y

  # ── Filtered data ──────────────────────────────────────────────────────────
  filtered <- reactive({
    req(input$year_range)
    d <- app_data

    if (!("All" %in% (input$region_sel %||% "All")) &&
        length(input$region_sel) > 0) {
      d <- d |> filter(region %in% input$region_sel)
    }

    d |>
      filter(year >= input$year_range[1], year <= input$year_range[2]) |>
      mutate(display_value = .data[[input$variable_sel %||% "value"]])

  }) |> bindEvent(input$apply_btn, ignoreNULL = FALSE)

  # ── Reset ──────────────────────────────────────────────────────────────────
  observeEvent(input$reset_btn, {
    updateSelectInput(session, "region_sel",   selected = "All")
    updateSliderInput(session, "year_range",   value    = c(2015, 2023))
    updateSelectInput(session, "variable_sel", selected = "value")
  })
```
